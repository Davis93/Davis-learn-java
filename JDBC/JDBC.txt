 2）获取一个数据库连接
 
  Connection conn=DriverManager.getConnection("连接url","用户名","密码");

Statement stmt=conn.createStatement();

stmt.executeUpdate("增，删，改");
or
ResultSet rs=stmt.executeQuery("查询SQL记录");
5.对结果进行处理哇
While(rs.next())
{
  //some operations
}

6.关闭连接
rs.close();
stmt.close();
conn.close();
